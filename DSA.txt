#include <stdio.h>
#define MAX 5 // Maximum size of stack

int stack[MAX];
int top = -1;

// Function to push elements
void push(int value) {
    if (top == MAX - 1) {
        printf("Stack Overflow! Cannot push %d\n", value);
    } else {
        top++;
        stack[top] = value;
        printf("Pushed %d to stack.\n", value);
    }
}

// Function to pop elements
void pop() {
    if (top == -1) {
        printf("Stack Underflow! Cannot pop from empty stack.\n");
    } else {
        printf("Popped %d from stack.\n", stack[top]);
        top--;
    }
}

// Function to display the stack
void display() {
    if (top == -1) {
        printf("Stack is empty.\n");
    } else {
        printf("Current stack: ");
        for (int i = 0; i <= top; i++) {
            printf("%d ", stack[i]);
        }
        printf("\n");
    }
}

int main() {
    // Trying to cause stack overflow
    push(10);
    push(20);
    push(30);
    push(40);
    push(50);
    push(60);  // Overflow here

    display();

    // Trying to cause stack underflow
    pop();
    pop();
    pop();
    pop();
    pop();
    pop();  // Underflow here

    return 0;
}
